12  public class MailingTest {
13      private static final String EMAIL = "foo@bar.uk";
14      private static final String NAME = "FooBarUK";
15      private static final String FIRSTNAME = "Deforest";
16      private static final String LANGUAGE = Locale.ENGLISH.getLanguage();
17  
18      @Test
19      public void objectCreationWithBuilder() {
20          Mailing mailing = Mailing.builder().email(EMAIL).surname(NAME).language(LANGUAGE).firstname(FIRSTNAME).build();
21  
22          assertThat(mailing).isNotNull();
23          assertThat(mailing.toString()).isNotNull();
24  
25          assertThat(mailing.getEmail()).isEqualTo(EMAIL);
26          assertThat(mailing.getSurname()).isEqualTo(NAME);
27          assertThat(mailing.getFirstname()).isEqualTo(FIRSTNAME);
28          assertThat(mailing.getLanguage()).isEqualTo("en");
29      }
30  
31      @Test
32      public void objectCreationWithoutBuilder() {
33          Mailing mailing = new Mailing(FIRSTNAME, NAME, EMAIL, LANGUAGE);
34  
35          assertThat(mailing).isNotNull();
36          assertThat(mailing.toString()).isNotNull();
37  
38          assertThat(mailing.getEmail()).isEqualTo(EMAIL);
39          assertThat(mailing.getSurname()).isEqualTo(NAME);
40          assertThat(mailing.getFirstname()).isEqualTo(FIRSTNAME);
41          assertThat(mailing.getLanguage()).isEqualTo(LANGUAGE);
42      }

44  }
